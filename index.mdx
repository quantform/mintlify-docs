# Quantform Documentation

Build with Quantform Core. Runtime tools, execution workflows, and guides for systematic trading developers.

## Developer quickstart

Make your first strategy run in minutes. Learn the Quantform runtime basics, process event streams, and validate behavior with replay windows.

<Card title="Get started" icon="arrow-right" href="/quickstart">
  Get Started ->

<CodeGroup>
```ts TypeScript
import { behavior, strategy, useExecutionMode, useLogger } from '@quantform/core';
import { interval, map, tap } from 'rxjs';

export default strategy(() => {
  behavior(() => {
    const { isReplay } = useExecutionMode();
    const { info } = useLogger('quickstart');

    return interval(1000).pipe(
      map(i => ({ timestamp: Date.now(), tick: i, mode: isReplay ? 'replay' : 'runtime' })),
      tap(event => info(`${event.mode} tick=${event.tick}`))
    );
  });

  return [];
});
```

```js JavaScript
const { behavior, strategy, useLogger } = require('@quantform/core');
const { interval, map, tap } = require('rxjs');

module.exports = strategy(() => {
  behavior(() => {
    const { info } = useLogger('quickstart');
    return interval(1000).pipe(
      map(i => ({ timestamp: Date.now(), tick: i })),
      tap(event => info(`tick=${event.tick}`))
    );
  });
  return [];
});
```
